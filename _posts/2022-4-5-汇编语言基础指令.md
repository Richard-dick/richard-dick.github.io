## 1 从hello world开始

任何语言的学习都应该发出对世界的声明。

```C
//hello.c
#include <stdio.h>
int main(void)
{
    printf(“Hello World\n”);
    return 0;
}
```

```
#hello.S
.section .data #数据
	output:	.ascii  "Hello World\n"
.section .text #汇编指导指令
.globl _start
_start: #机器指令助记符
	movl	$4, %eax
	movl	$1, %ebx
	movl	$output, %ecx
	movl	$12, %edx
	int	$0x80
	movl	$1, %eax
	movl	$0, %ebx
	int	$0x80
```

从`_start`开始，其上为数据区，其下为代码区

见output行：

    output:	.ascii  "Hello World\n"

可以延伸出一些常用的数据类型：

### 1.1 常用数据类型

- 主要的整数类型
  - .byte  -- 字节整数
  - .word  -- 字整数（2字节）
  - .int  -- 整数（4字节）
  - .long  -- 长整数（4字节）
  - .quad  -- 8字节整数
  - .octa  -- 16字节整数

- 字符串类型
  - .ascii  -- 字符串（非\null结尾）
  - .asciz  -- \null结尾的字符串


我们再看movl指令，它涉及到不同的数据类型，需要后缀区分。分'b','w','l'，和上面的数据类型一致。

其中有涉及到\$号，对于数字来收，代表立即数；而对symbol来说，代表地址。

AT&T的mov是从前往后，Intel反过来。需要注意的是，内存操作数最多只有一个。也就是使用地址的操作数。

此外，mov还有扩展功能，比如小位数向多位数送；需要扩展

- movzx，zero-extension
- movsx，sign-extension


### 1.2 一般指令序列功能

对于int $0x80（Linux-32系统调用）来说

- EAX：系统调用号，4号是写文件，1号是程序退出
- EBX、ECX、EDX、…：传递参数
- 4号调用：写文件
    - EBX：文件描述符，1表示终端
    - ECX：输出缓冲区（Buffer）的地址
    - EDX：输出的字节数
- 1号调用：程序退出
  - EBX：返回值

所以最后一段的意思是

  movl	$4, %eax        # 准备写文件
	movl	$1, %ebx        # 写到终端
	movl	$output, %ecx   # 写的首地址
	movl	$12, %edx       # 写的长度
	int	$0x80               # 软中断 处理上面
	movl	$1, %eax        # 准备退出
	movl	$0, %ebx        # 主函数返回值
	int	$0x80               # 软中断，处理上面的退出


### 1.3 Linux-64

对于64位来说，情况稍有不同，不过可以用同一个程序编译一下看看调用号：

1. 系统调用指令不同
   - syscall 代替 int $0x80
2. 系统调用号不同
    - 1号为写文件，60号为程序退出 
3. 参数寄存器不同
    - eax、ebx、ecx、edx…替换为rax、rdi、rsi、rdx…


### 1.4 AT&T vs Intel

它们最大的一个区别是%，\$号和目的寄存器的位置；

AT&T在右边，Intel在左边


```
#hello.S
        .intel_syntax noprefix
.section .data
        output: .ascii  "Hello World\n"
.section .text
.globl _start
_start:
/* output  like printf */
        mov     ecx, offset output
        mov     eax, 4
        mov     ebx, 1
        mov     edx, 12
        int     0x80
/* exit */
        mov     eax, 1
        mov     ebx, 0
        int     0x80
```

说到mov，就必须得提到寻址方式：

### 1.5 内存操作数寻址方式

表示如下：

disp(基址，指标，范围) = 基址+指标*范围+disp；

比如4(%ebx, %esi, 2), 就是EA = (%ebx)+4+2*(%esi).

> esp不能做index

另外一个比较特殊的就是数据区，

- $symbol是地址
- symbol又是内容
- symbol+num又是地址。。。


## 2 输出0123456789

```
#loop1.S
.section .data
  bytevar:
	.rept	10
  	.byte 	0
  	.endr
  	.byte 	13, 10
.section .text
.globl _start
_start:
  mov 	$10, %ecx
  mov 	$0x30, %al
  mov 	$bytevar, %ebx
L1:
  movb 	%al, (%ebx)
  inc  	%ebx
  inc  	%al
  loop 	L1
#output
 mov   $4, %eax
  mov   $1, %ebx
  mov   $bytevar, %ecx
  mov   $12, %edx
  int   $0x80
#exit
  movl  $1, %eax
  movl  $0, %ebx
  int   $0x80
```

### 2.1 重复声明

	.rept	10
  	.byte 	0
  	.endr

能够声明10个byte，语法简洁明了，不做解释

### 2.2 loop相对跳转

loop自带一个计数寄存器。

上述代码种，%ecx就是其计数寄存器。大意就是循环10次，每次加一存入。

### 2.3 读写文件

先上图：

![读写文件](https://raw.githubusercontent.com/Richard-dick/richard-dick.github.io/main/_posts/inc/AS5.png)

我们试一下在终端输入：

```
#loop2.S
.section .data
  bytevar:
        .rept 10
        .byte 0
        .endr
        .byte 13, 10
  charbuf: .byte 0, 0 # 缓冲字符
.section .text
.globl _start
_start:
#input 2 char
  mov   	$3, %eax # 要输入
  mov   	$1, %ebx # 在终端
  mov   	$charbuf, %ecx # 输入保存位置
  mov   	$2, %edx # 输入长度

  mov   	$5, %ecx # 开loop
  mov   	$bytevar, %ebx
  movw  	charbuf, %ax
L1:
  movb  	%al, (%ebx)

#output
  mov   	$4, %eax
  mov   	$1, %ebx
  mov   	$bytevar, %ecx
  mov   	$12, %edx
  int   	$0x80

  movl  	$1, %eax
  movl  	$0, %ebx
  int   	$0x80
```


## 3 跳转表程序

本节用一个跳转表程序来讲解：

```C
int switch_case(int x, int y)
{
  int result;

  switch(x)
  {
    case 0: result = y+1;
            break;
    case 1: result = y+2;
            break;
    case 2: result = y+4;
            break;
    case 3: result = y+8;
            break;
    case 4: result = y+16;
            break;
    default: result = y+y;
  }
  return result;
}
```

```
.text
switch_case:
        cmpl    $4, 8(%ebp)
        ja      .L2 # 如果后者大，above
        movl    8(%ebp), %eax
        sall    $2, %eax # 左移两位对齐，这里是因为.L4种都是.long
        addl    $.L4, %eax # 将x作为偏移量，所以必须是连续的
        movl    (%eax), %eax
        jmp     *%eax # 点明是间接跳转，防止其为符号时混淆。
        .section        .rodata # 只读数据区，汇编器会自动将其提前。
        .align 4
        .align 4
.L4:
        .long   .L3
        .long   .L5
        .long   .L6
        .long   .L7
        .long   .L8
.text
.L3:
        movl    12(%ebp), %eax
        addl    $1, %eax
        movl    %eax, -4(%ebp)
        jmp     .L9
.L5:
        movl    12(%ebp), %eax
        addl    $2, %eax
        movl    %eax, -4(%ebp)
        jmp     .L9
```

### 3.1 switch 建表

注意.L4。它相当于在内存中建表，所以可以通过对.L4符号的偏移来完成跳转，看起来可能没有那么花哨。

之前的一端是把.L4的地址拿出来，然后通过地址读取其中的内容。这是对于符号而言的操作，为了得到作为内容的跳转地址而为的操作。

### 3.2 jcc跳转指令

这里很直接，是后者相对前者而言的。

above,below是相对无符号数而言；

greater,less是相对有符号数而言。

n否定，e相等。

> 汇编语言一大特点是顺序执行，所以跳转一定要确定该不该加。

### 3.3 翻转程序

```
#revstring.S
.section .data
  inputbuf:
        .rept 80
        .byte 0
        .endr
  outputbuf:
        .rept 80
        .byte 0
        .endr
.section .text
.globl _start
_start:
#input string
  mov           	$3, %eax
  mov           	$1, %ebx
  mov           	$inputbuf, %ecx
  mov           	$80, %edx
  int           	$0x80
#strlen
  mov           	$inputbuf, %ebx
  xor          	  %ecx, %ecx # 这里给一个异或清零0
L1:
  movb          	(%ebx), %al
  cmp           	$10, %al # 这里是和'\n'比，确保只输入一行。
  je            	L2
  inc           	%ebx
  inc           	%ecx
  jmp           	L1
#reverse
L2:
  push          	%ecx
  dec           	%ebx
  mov           	$outputbuf, %edx
  test          	%ecx, %ecx
  jz            	L4
L3:
  movb          	(%ebx), %al
  movb          	%al, (%edx)
  dec           	%ebx
  inc           	%edx
  dec           	%ecx
  jnz           	L3
L4:
  movb          	$10, (%edx)
  inc        %edx
  sub       $outputbuf, %edx
#output
  pop           	%edx
  inc           	%edx
  mov           	$4, %eax
  mov           	$1, %ebx
  mov           	$outputbuf, %ecx
  int           	$0x80
#exit
  movl          	$1, %eax
  movl          	$0, %ebx
  int           	$0x80
```

从这里我们可以看见，写一个功能稍微多一点的程序就如此费劲了。
